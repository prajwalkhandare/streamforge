🔧 USEFUL COMMANDS
------------------
# Get pods status
kubectl get pods -n {{ .Release.Namespace }} -w

# Check logs
kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/component=backend
kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/component=frontend

# Scale backend
kubectl scale -n {{ .Release.Namespace }} deployment/{{ .Values.backend.name }} --replicas=5

# Port forward to database
kubectl port-forward -n {{ .Release.Namespace }} svc/postgresql 5432:5432
